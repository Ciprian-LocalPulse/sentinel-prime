NAVIGATION MECHANISM — showTab()

All 11 tab panels exist in the DOM simultaneously as .tab-pane elements. Visibility is controlled via CSS class toggling — a pure-CSS, zero-latency navigation pattern with lazy chart initialisation:

// ── Core navigation function ────────────────────────────── function showTab(name) { // Hide all panels document.querySelectorAll('.tab-pane').forEach(p => p.classList.remove('active')); // Remove active state from all sidebar items document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active')); // Show the target panel const tab = document.getElementById(`tab-${name}`); if (tab) tab.classList.add('active'); // Activate matching sidebar item document.querySelectorAll('.nav-item').forEach(n => { if (n.getAttribute('onclick')?.includes(name)) n.classList.add('active'); }); // Lazy-initialise charts only when the tab is first viewed initChartsForTab(name); } // ── Lazy chart initialisation map ───────────────────────── function initChartsForTab(name) { if (name === 'dashboard') { initAlertTrendChart(); initSevPieChart(); } if (name === 'vulnerabilities') { initPatchChart(); } if (name === 'network') { initNetworkChart(); initFirewallChart(); } if (name === 'ueba') { initUebaChart(); } if (name === 'reports') { initMaturityChart(); initIncidentTrendChart(); } }

CHART INSTANCE REGISTRY

All Chart.js instances are stored in a global registry object to enable targeted updates and safe destruction before re-initialisation:

// Global chart registry — prevents Chart.js canvas reuse errors const chartInstances = {}; function destroyChart(id) { if (chartInstances[id]) { try { chartInstances[id].destroy(); } catch(e) {} delete chartInstances[id]; } } // Usage pattern — always destroy before re-init function initAlertTrendChart() { const id = 'alertTrendChart'; destroyChart(id); // safe destroy chartInstances[id] = new Chart(document.getElementById(id), { /* config */ }); }

Data Binding Architecture
& Variable Reference

Every data element in Sentinel Prime is bound to a named JavaScript variable. Replacing mock data with real API responses requires only identifying the variable name and calling the appropriate update function. No rebuilding, no recompiling.

CHART.JS DATASET UPDATE PATTERN

// Universal chart refresh pattern — works for ALL 8 chart instances async function refreshChartData(chartId, apiEndpoint, authToken) { try { const res = await fetch(apiEndpoint, { headers: { 'Authorization': `Bearer ${authToken}`, 'Content-Type': 'application/json' } }); if (!res.ok) throw new Error(`HTTP ${res.status}`); const data = await res.json(); const chart = chartInstances[chartId]; if (!chart) return; chart.data.labels = data.labels; chart.data.datasets[0].data = data.values; chart.update('active'); // smooth animated transition } catch (err) { console.error(`[SENTINEL] Chart ${chartId} refresh failed:`, err); // Falls back to last known good data — no visual disruption } } // Refresh alert trend every 30 seconds setInterval(() => refreshChartData('alertTrendChart', '/api/alerts/trend', AUTH_TOKEN), 30000);

KPI COUNTER ANIMATED UPDATE

// Animated count-up for KPI values — visual emphasis on change function updateKPI(elementId, newValue, prevValue = 0) { const el = document.getElementById(elementId); if (!el) return; const diff = newValue - prevValue; const steps = 30; const step = diff / steps; let current = prevValue; let count = 0; const timer = setInterval(() => { current += step; count++; el.textContent = Math.round(current).toLocaleString(); if (count >= steps) { el.textContent = newValue.toLocaleString(); clearInterval(timer); } }, 20); // Flash colour on change el.style.color = diff > 0 ? 'var(--red)' : 'var(--green)'; setTimeout(() => el.style.color = '', 3000); } // Usage updateKPI('alertCount', 27, 23); // topbar badge — 23 → 27

THREAT TABLE ROW INJECTION

// Prepend new threat event to live feed table (Module 01 / Module 02) function prependThreatRow({ timestamp, severity, sourceIP, indicator, mitre, status }) { const tbody = document.getElementById('threat-feed-body'); if (!tbody) return; const sevClass = { CRITICAL: 'sev-critical', HIGH: 'sev-high', MEDIUM: 'sev-medium', LOW: 'sev-low' }; const sevColour = { CRITICAL: 'var(--red)', HIGH: 'var(--amber)', MEDIUM: 'var(--cyan)', LOW: 'var(--green)' }; const row = document.createElement('tr'); row.innerHTML = ` <td style="color:var(--text-muted)">${timestamp}</td> <td><span class="sev ${sevClass[severity] || ''}">${severity}</span></td> <td>${sourceIP}</td> <td>${indicator}</td> <td style="color:var(--text-muted)">${mitre}</td> <td style="color:${sevColour[severity] || 'var(--text-primary)'}">${status}</td> `; row.style.background = 'rgba(0,212,255,0.06)'; tbody.insertBefore(row, tbody.firstChild); while (tbody.rows.length > 50) tbody.deleteRow(-1); setTimeout(() => row.style.background = '', 2000); }

Real-Time WebSocket
Stream Manager

/* ═══════════════════════════════════════════════════════ SentinelWSManager — drop-in real-time stream manager Replaces all setInterval polling with live push events ═══════════════════════════════════════════════════════ */ class SentinelWSManager { constructor(wsUrl, authToken) { this.url = wsUrl; this.token = authToken; this.ws = null; this.reconnectDelay = 3000; this.maxRetries = 10; this.retryCount = 0; this.connect(); } connect() { this.ws = new WebSocket(`${this.url}?token=${this.token}`); this.ws.onopen = () => { this.retryCount = 0; console.log('[SENTINEL WS] Stream connected'); updateTopbarStatus('green', 'SIEM LIVE'); }; this.ws.onmessage = ({ data }) => { try { this.dispatch(JSON.parse(data)); } catch(e) { console.warn('[SENTINEL WS] Bad message', e); } }; this.ws.onclose = () => { updateTopbarStatus('amber', 'RECONNECTING'); if (this.retryCount < this.maxRetries) { this.retryCount++; setTimeout(() => this.connect(), this.reconnectDelay * this.retryCount); } else { updateTopbarStatus('red', 'STREAM OFFLINE'); } }; this.ws.onerror = () => updateTopbarStatus('red', 'STREAM ERROR'); } dispatch(msg) { switch(msg.type) { case 'alert': prependThreatRow(msg.payload); updateKPI('alertCount', +document.getElementById('alertCount').textContent + 1); break; case 'incident': renderIncidentQueue(msg.payload); break; case 'network_stats': pushNetworkPoint(msg.payload); break; case 'chart_update': const c = chartInstances[msg.chartId]; if (c) { c.data.labels = msg.labels; c.data.datasets[0].data = msg.values; c.update('active'); } break; case 'kpi_update': Object.entries(msg.kpis).forEach(([id,val]) => updateKPI(id, val)); break; default: console.warn(`[SENTINEL WS] Unknown type: ${msg.type}`); } } send(payload) { if (this.ws?.readyState === WebSocket.OPEN) this.ws.send(JSON.stringify(payload)); } close() { this.maxRetries = 0; this.ws?.close(); } } // Initialise const sentinelStream = new SentinelWSManager( 'wss://siem.enterprise.io/realtime', 'YOUR_WS_AUTH_TOKEN' );

// Push a new data point with sliding window (keeps last N points) function pushNetworkPoint({ timestamp, ingress_gb, egress_gb }, maxPoints = 24) { const chart = chartInstances['networkChart']; if (!chart) return; chart.data.labels.push(timestamp); chart.data.datasets[0].data.push(ingress_gb); chart.data.datasets[1].data.push(egress_gb); if (chart.data.labels.length > maxPoints) { chart.data.labels.shift(); chart.data.datasets.forEach(ds => ds.data.shift()); } chart.update('none'); // 'none' = no animation (faster for live data) }
